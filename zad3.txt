#include <iostream>
using namespace std;
int main()
{
    int tab[10][10],i,j;
      for (i = 0; i < 9;i++){
        for (j = 0; j < 9;j++){
            if(i == j){

            tab[i][j] = 1;
            }
            else {
                tab[i][j] = 0;
            }
        }
    }
    for (i = 0; i < 9; i++) {

        for (j = 0; j < 9; j++) {

            cout << tab[i][j] << " ";
        }
        cout << endl;
    }
    cout << endl;
}
